Powershell used to capture screen data

# set directory to the Tesseract OCR folder
cd 'C:\Program Files\Tesseract-OCR\'
# import
Add-Type -AssemblyName System.Drawing
# setup bitmap with size
$Bitmap = New-Object System.Drawing.Bitmap 450, 290 # or 190

# capture screenspace
$Graphics = [System.Drawing.Graphics]::FromImage($Bitmap)
$Graphics.CopyFromScreen(10, 100, 0, 0, $Bitmap.Size)

# save to memorystream
$MemoryStream = New-Object System.IO.MemoryStream
$Bitmap.Save($MemoryStream, [System.Drawing.Imaging.ImageFormat]::Png)
$Bytes = $MemoryStream.ToArray()

# ocr with input from memory stream cause you can't just pipe it -3-
$psi = New-Object System.Diagnostics.ProcessStartInfo
$psi.FileName = 'C:\Program Files\Tesseract-OCR\tesseract.exe'
$psi.Arguments = 'stdin stdout -l eng'
$psi.RedirectStandardInput = $true
$psi.RedirectStandardOutput = $true
$psi.UseShellExecute = $false
$psi.CreateNoWindow = $true

# run the command
$p = [System.Diagnostics.Process]::Start($psi)
$p.StandardInput.BaseStream.Write($Bytes, 0, $Bytes.Length)
$p.StandardInput.BaseStream.Close()
$result = $p.StandardOutput.ReadToEnd()

# message data to be sent to ahk
$messageString = "Data:{'$result'}:ataD"

$p.WaitForExit()

# cleanup memory
$Graphics.Dispose()
$Bitmap.Dispose()
$MemoryStream.Flush()
$MemoryStream.Dispose()
$p.Dispose()

# send message back to script
$WM_COPYDATA = 0x4a

# import
$Signature = @"
[DllImport("user32.dll")]public static extern IntPtr SendMessage(IntPtr hWnd, int Msg, IntPtr wParam, ref COPYDATASTRUCT lParam); public struct COPYDATASTRUCT { public IntPtr dwData; public int cbData; public IntPtr lpData;}
"@
$WinAPI = Add-Type -MemberDefinition $Signature -Name "Win32SendMessage" -Namespace "Win32" -PassThru

# encode and send message to ahk
$bytes = [System.Text.Encoding]::Unicode.GetBytes($messageString)
$ptr = [System.Runtime.InteropServices.Marshal]::AllocHGlobal($bytes.Length)
[System.Runtime.InteropServices.Marshal]::Copy($bytes, 0, $ptr, $bytes.Length)
$cds = [Win32.Win32SendMessage+COPYDATASTRUCT]::new()
$cds.dwData = [IntPtr]::Zero
$cds.cbData = $bytes.Length
$cds.lpData = $ptr
$targetHwnd = (Get-Process | Where-Object { $_.MainWindowTitle -like "easter.egg.pathing"}).MainWindowHandle
if ($targetHwnd) {[Win32.Win32SendMessage]::SendMessage($targetHwnd, $WM_COPYDATA, [IntPtr]::Zero, [ref]$cds)}
# cleanup
[System.Runtime.InteropServices.Marshal]::FreeHGlobal($ptr)
